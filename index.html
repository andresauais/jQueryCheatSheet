Functions and Selectors

Create

Javascript

var p = document.createElement("p");
p.innerHTML = "This is a new paragraph"
body.appendChild(p);

jQuery

var p = $('<p>');
p.text("This is a new paragraph") $("body").append(p);

Remove

Javascript

document.createElementByID("p").remove();

jQuery

$('p').remove();

Append

Javascript

body.append(p);

jQuery

$("body").append(p);

Prepend

Javascript

document.getElementById("p").insertBefore("Random Element");

jQuery

$("p").prepend("Prepended text");

Fade in

Javascript

document.getElementById('myBtn').onclick = function () { slideSource.classList.toggle('fade'); }

Css classes

#slideSource { opacity: 1; transition: opacity 1s; } #slideSource.fade { opacity: 0; }

jQuery

$("button").click(function(){ $("#div1").fadeIn(); $("#div2").fadeIn("slow"); $("#div3").fadeIn(3000); });
